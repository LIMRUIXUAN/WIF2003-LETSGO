const { test } = require('node:test');
const assert = require('node:assert/strict');
const Destination = require('../models/Destination');
const User = require('../models/User');
const Trip = require('../models/Trip');

test('Destination Schema validations', async (t) => {
  await t.test('Validates clean with proper schema inputs', () => {
    const destination = new Destination({
      id: 1,
      name: 'Rainforest Retreat',
      location: 'Taman Negara',
      category: 'hotel',
      ecoScore: 9.3
    });
    const error = destination.validateSync();
    assert.equal(error, undefined);
  });

  await t.test('Rejects invalid category and out-of-range ecoScore', () => {
    const destination = new Destination({
      id: 2,
      name: 'Rainforest Retreat',
      location: 'Taman Negara',
      category: 'invalid-category', 
      ecoScore: 15 // Exceeds max
    });
    const error = destination.validateSync();
    assert.ok(error.errors.category, 'Should reject invalid category');
    assert.ok(error.errors.ecoScore, 'Should reject ecoScore > 10');
  });
});

test('Trip Schema validations', async (t) => {
  await t.test('Validates start and end date logic', () => {
    const trip = new Trip({
      userEmail: 'demo@ecoplanner.com',
      name: 'UM Break Trip',
      startDate: new Date('2026-06-01'),
      endDate: new Date('2026-05-20') // Invalid: End date before start date
    });
    
    // Assuming you have a custom validator in your Trip schema for this
    const error = trip.validateSync();
    assert.ok(error, 'Should fail validation due to chronologically incorrect dates');
  });

#### **SECTION 3 BACK-END**

**3.2 Approaches/Techniques used to improve quality of Web application** 
To ensure high software quality, performance, and usability within the EcoTravel Planner, our team implemented a rigorous, lightweight testing architecture.

* **Native Backend Testing:** Instead of relying on heavy frameworks like Jest or spinning up live MongoDB instances, we utilized Node.js native `node:test` runner. By leveraging Mongoose’s built-in `.validateSync()`, we synchronously tested schema integrity, reducing test execution time to under 100ms.
* **API Stubbing:** For API route testing, we utilized dynamic Express servers (`app.listen(0)`) and stubbed database operations. This prevented environment pollution and avoided database connection timeouts, significantly improving the backend's reliability before deployment.
* **UI/Navigation State Management:** On the client side, we conducted systematic UI testing to verify secure navigation. This included ensuring protected routes accurately detect JWT tokens/local storage (`ecoUserEmail`) and restrict unauthorized access, thereby enhancing the overall security of the platform.

**3.3 Error handling** 

* **Synchronous Schema Validation:** We implemented strict error handling at the database model level. By defining required fields, enums (e.g., categories must be 'hotel', 'nature', etc.), and constraints (e.g., `ecoScore` max limit), any malformed data attempting to enter the database is immediately caught. The schema returns an error object detailing the exact failing property, which our Express error handling middleware (`middleware/errorHandler.js`) intercepts and returns to the frontend as a clean 400 Bad Request JSON response.
* **Frontend Input Validation:** Error handling is also implemented on client-side modules, such as the CO2 Calculator Modal. If a user inputs a goal below the minimum threshold (e.g., < 10 kg), the application prevents the save operation and triggers a toast warning ("Enter a goal between 10 and 9999 kg"), preventing invalid database calls and guiding the user to correct their input.

#### **SECTION 4 DATABASE**

**4.1 Database Design** The database design leverages MongoDB, implemented via Mongoose schemas to provide a rigid structure to our document collections.

* **Destination Schema:** Utilizes schema alias mapping to expose canonical properties (e.g., mapping `eco` to `ecoScore`) and defines unique indexes on the `id` field. We also implemented text search indexing on `name` and `location` to optimize the performance of the Explore page search functionality.
* **User Schema:** Contains validation constraints to reject malformed email formats and ensures the `favorites` array only accepts valid integer IDs referencing destination documents.
